## 流程

```shell
nix shell nixpkgs#k3sup

# 主节点
k3sup install --user shelken --ip 192.168.6.110 \
  --k3s-extra-args "--disable traefik --disable servicelb" --ssh-key ~/.ssh/id_ed25519

# Install K3s joining a cluster as an agent
k3sup join \
    --user shelken \
    --ip 192.168.6.111 \
    --server-ip 192.168.6.110 \
    --server-user shelken \
    --ssh-key ~/.ssh/id_ed25519

```
